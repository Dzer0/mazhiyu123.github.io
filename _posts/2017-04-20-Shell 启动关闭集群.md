#### 启动集群

第一个版本
```
#!/bin/bash
#启动集群  在master1上运行

zkServer.sh start


ssh master2 << EOF
zkServer.sh start
hadoop-daemon.sh start zkfc
mr-jobhistory-daemon.sh start historyserver
EOF

ssh slave1 << EOF
zkServer.sh start
EOF

hadoop-daemon.sh start zkfc
start-dfs.sh
start-yarn.sh

```

#### 关闭集群

第一个版本
```
#!/bin/bash
#停止集群 在master1节点上运行

stop-yarn.sh
stop-dfs.sh
hadoop-daemon.sh stop zkfc
zkServer.sh stop


ssh master2 <<EOF
zkServer.sh stop
hadoop-daemon.sh stop zkfc
mr-jobhistory-daemon.sh stop historyserver
EOF

ssh slave1 <<EOF
zkServer.sh stop
EOF
```
